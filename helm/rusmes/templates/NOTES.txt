==> Access Instructions

1. Get the application URL:

{{ if .Values.ingress.enabled -}}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "rusmes.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "rusmes.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "rusmes.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.ports.jmap }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "rusmes.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Monitor deployment status:

  kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "rusmes.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -w

3. View logs:

  kubectl logs --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "rusmes.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -f

{{ if .Values.config.metrics.enabled -}}
4. Access metrics:

  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "rusmes.fullname" . }} 9090:9090
  # Then visit http://localhost:9090/metrics
{{- end }}

==> Next Steps

{{ if .Values.postgresql.enabled -}}
1. Set up database schema:
   kubectl exec -it --namespace {{ .Release.Namespace }} {{ include "rusmes.fullname" . }}-0 -- rusmes-cli migrate
{{- end }}

2. Create admin user:
   kubectl exec -it --namespace {{ .Release.Namespace }} {{ include "rusmes.fullname" . }}-0 -- rusmes-cli user create admin@{{ .Values.config.hostname }} --admin

3. Configure DNS records for your domain:
   - MX record pointing to {{ .Values.config.hostname }}
   {{ if .Values.config.security.enableDkim -}}
   - DKIM TXT record (run: rusmes-cli dkim generate {{ .Values.config.hostname }})
   {{- end }}
   {{ if .Values.config.security.enableSpf -}}
   - SPF TXT record (example: "v=spf1 mx ~all")
   {{- end }}
   {{ if .Values.config.security.enableDmarc -}}
   - DMARC TXT record (example: "v=DMARC1; p=quarantine; rua=mailto:postmaster@{{ .Values.config.hostname }}")
   {{- end }}

{{ if .Values.tls.certManager.enabled -}}
4. TLS certificates will be automatically provisioned via cert-manager
{{- else }}
4. Configure TLS certificates manually in the secret: {{ .Values.tls.existingSecret | default (printf "%s-tls" (include "rusmes.fullname" .)) }}
{{- end }}

{{ if .Values.backup.enabled -}}
5. Verify backup configuration:
   kubectl get cronjob --namespace {{ .Release.Namespace }} {{ include "rusmes.fullname" . }}-backup
{{- end }}
